#### 3.2.1 GLM API调用封装 (core/agent/chat_client.py)

```python
import requests
from typing import Optional
from config.settings import AppConfig

class ChatClient:
    def __init__(self, api_key: str, model: str = "glm-4"):
        self.api_key = api_key
        self.model = model
        self.base_url = "https://open.bigmodel.cn/api/paas/v4"

    def chat(self, messages: list, temperature: float = 0.7) -> str:
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        payload = {
            "model": self.model,
            "messages": messages,
            "temperature": temperature
        }
        response = requests.post(
            f"{self.base_url}/chat/completions",
            headers=headers,
            json=payload,
            timeout=30
        )
        response.raise_for_status()
        return response.json()["choices"][0]["message"]["content"]

    def analyze_note(self, note_content: str) -> dict:
        prompt = f"""请分析以下算法心得，提取关键信息：
        {note_content}

        请以JSON格式返回，包含：
        - algorithm_type: 算法类型（如：动态规划、贪心、DFS等）
        - key_points: 关键秘术列表
        - difficulty: 难度等级（简单/中等/困难）
        - tags: 相关标签列表
        - summary: 一句话总结"""
        messages = [{"role": "user", "content": prompt}]
        result = self.chat(messages)
        return self._parse_json_response(result)

    def generate_questions(self, note_content: str, question_types: list, count: int = 3) -> list:
        prompt = f"""根据以下算法心得，生成{count}道试炼：
        {note_content}

        试炼类型要求：{', '.join(question_types)}

        请为每道题包含：
        - question_type: 试炼类型
        - content: 题目内容
        - answer: 参考答案
        - explanation: 解析"""
        messages = [{"role": "user", "content": prompt}]
        result = self.chat(messages)
        return self._parse_json_response(result)

    def evaluate_answer(self, question: str, user_answer: str, correct_answer: str) -> dict:
        prompt = f"""试炼：{question}
        用户答案：{user_answer}
        参考答案：{correct_answer}

        请分析用户答案，给出：
        - is_correct: 是否正确
        - feedback: 详细评价
        - improvement: 改进建议"""
        messages = [{"role": "user", "content": prompt}]
        result = self.chat(messages)
        return self._parse_json_response(result)

    def _parse_json_response(self, response: str) -> dict:
        import json
        import re
        json_match = re.search(r'\{[\s\S]*\}|\[[\s\S]*\]', response)
        if json_match:
            return json.loads(json_match.group())
        return {"raw": response}
```

#### 3.2.4 薄弱点分析器 (core/agent/weak_point_analyzer.py)

```python
from typing import List, Dict
from collections import defaultdict
from data.repositories import AnswerRecordRepository, NoteRepository

class WeakPointAnalyzer:
    def __init__(self,
                 answer_repo: AnswerRecordRepository,
                 note_repo: NoteRepository):
        self.answer_repo = answer_repo
        self.note_repo = note_repo

    def analyze(self, days: int = 30) -> Dict:
        recent_records = self.answer_repo.get_recent_records(days)

        type_stats = defaultdict(lambda: {"total": 0, "correct": 0})
        for record in recent_records:
            question = record.question
            note = question.note
            type_stats[note.algorithm_type]["total"] += 1
            if record.is_correct:
                type_stats[note.algorithm_type]["correct"] += 1

        weak_points = []
        for algo_type, stats in type_stats.items():
            if stats["total"] >= 3:
                accuracy = stats["correct"] / stats["total"]
                if accuracy < 0.7:
                    weak_points.append({
                        "type": algo_type,
                        "accuracy": accuracy,
                        "total_questions": stats["total"],
                        "correct_questions": stats["correct"]
                    })

        weak_points.sort(key=lambda x: x["accuracy"])

        return {
            "weak_points": weak_points,
            "overall_accuracy": self._calc_overall_accuracy(type_stats),
            "recommendations": self._generate_recommendations(weak_points)
        }

    def _calc_overall_accuracy(self, type_stats: Dict) -> float:
        total = sum(s["total"] for s in type_stats.values())
        correct = sum(s["correct"] for s in type_stats.values())
        return correct / total if total > 0 else 0.0

    def _generate_recommendations(self, weak_points: List[Dict]) -> List[str]:
        recommendations = []
        for wp in weak_points[:3]:
            recommendations.append(
                f"建议加强{wp['type']}的修习，当前胜率仅{wp['accuracy']:.1%}"
            )
        return recommendations
```

### 3.3 遗忘曲线模块 (core/memory/forgotten_curve.py)

```python
from datetime import datetime, timedelta
from typing import List
from config.settings import AppConfig

class ForgottenCurve:
    """
    基于艾宾浩斯遗忘曲线的修炼调度算法
    """

    def __init__(self, intervals: List[int] = None):
        self.intervals = intervals or AppConfig().REVIEW_INTERVALS

    def calculate_next_review(self,
                              mastery_level: int,
                              review_count: int,
                              last_review_date: datetime) -> datetime:
        """
        根据领悟程度和修炼次数计算下次修炼时间

        Args:
            mastery_level: 领悟程度 0-100
            review_count: 已修炼次数
            last_review_date: 上次修炼时间

        Returns:
            下次修炼时间
        """
        if mastery_level >= 90:
            interval = self.intervals[-1] * 2
        elif mastery_level >= 70:
            idx = min(review_count, len(self.intervals) - 1)
            interval = self.intervals[idx] * 1.5
        elif mastery_level >= 50:
            idx = min(review_count, len(self.intervals) - 1)
            interval = self.intervals[idx]
        else:
            interval = self.intervals[0]

        return last_review_date + timedelta(days=int(interval))

    def get_review_schedule(self, note_id: int, created_at: datetime) -> List[dict]:
        """
        获取某个心得的完整修炼计划

        Args:
            note_id: 心得ID
            created_at: 心得创建时间

        Returns:
            修炼计划列表
        """
        schedule = []
        current_date = created_at

        for i, interval in enumerate(self.intervals):
            review_date = current_date + timedelta(days=interval)

            schedule.append({
                "note_id": note_id,
                "review_date": review_date,
                "interval": interval,
                "is_key_review": interval in [7, 14, 30]
            })
            current_date = review_date

        return schedule

    def update_mastery_level(self,
                             current_level: int,
                             is_correct: bool,
                             difficulty: str) -> int:
        """
        根据应战情况更新领悟程度

        Args:
            current_level: 当前领悟程度
            is_correct: 应战是否正确
            difficulty: 试炼难度

        Returns:
            新的领悟程度
        """
        difficulty_factor = {
            "简单": 5,
            "中等": 10,
            "困难": 15
        }.get(difficulty, 10)

        if is_correct:
            increment = difficulty_factor * (1 + current_level / 100)
            new_level = min(100, current_level + increment)
        else:
            decrement = difficulty_factor * 0.5
            new_level = max(0, current_level - decrement)

        return int(new_level)
```

### 3.4 修炼调度模块 (core/scheduler/review_scheduler.py)

```python
from datetime import datetime, timedelta
from typing import List, Optional
from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.triggers.cron import CronTrigger
from data.repositories import NoteRepository, ReviewRecordRepository
from data.models import Note
from core.memory.forgotten_curve import ForgottenCurve
from core.scheduler.email_sender import EmailSender
from config.settings import AppConfig

class ReviewScheduler:
    def __init__(self,
                 config: AppConfig,
                 note_repo: NoteRepository,
                 review_repo: ReviewRecordRepository,
                 email_sender: EmailSender,
                 forgotten_curve: ForgottenCurve):
        self.config = config
        self.note_repo = note_repo
        self.review_repo = review_repo
        self.email_sender = email_sender
        self.forgotten_curve = forgotten_curve
        self.scheduler = BackgroundScheduler()

    def start(self):
        if self.config.REVIEW_ENABLED:
            hour, minute = map(int, self.config.REVIEW_TIME.split(":"))
            self.scheduler.add_job(
                self.send_daily_review_email,
                CronTrigger(hour=hour, minute=minute),
                id="daily_review_email",
                replace_existing=True
            )
            self.scheduler.start()

    def stop(self):
        self.scheduler.shutdown()

    def send_daily_review_email(self):
        today = datetime.now().date()
        notes_due = self.note_repo.get_notes_due_for_review(today)

        if not notes_due:
            return

        email_content = self._build_email_content(notes_due)
        self.email_sender.send(email_content)

        for note in notes_due:
            self._create_review_record(note, today)

    def _build_email_content(self, notes: List) -> dict:
        subject = f"📚 算法修习助手 - 今日修炼提醒"

        overview = f"今日修炼目标：{len(notes)}个秘术\n"
        overview += f"预计用时：{len(notes) * 8}分钟\n\n"

        details = ""
        for i, note in enumerate(notes, 1):
            details += f"{'━' * 40}\n"
            details += f"📖 秘术 {i}：{note.title}\n"
            details += f"{'━' * 40}\n"
            details += f"{note.content}\n\n"

        body = f"亲爱的冒险者，您好！\n\n{overview}\n{details}"
        body += "\n坚持修炼，打卡修习！"

        return {
            "subject": subject,
            "body": body
        }

    def _create_review_record(self, note: Note, review_date: datetime):
        self.review_repo.create(
            note_id=note.id,
            review_date=review_date,
            status="pending"
        )

    def get_upcoming_reviews(self, days: int = 7) -> List[dict]:
        end_date = datetime.now().date() + timedelta(days=days)
        notes = self.note_repo.get_notes_due_for_review(end_date)

        schedule = []
        for note in notes:
            schedule.append({
                "note": note,
                "review_date": note.next_review_date
            })
        return schedule
```

### 3.5 邮件发送模块 (core/scheduler/email_sender.py)

```python
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from typing import Dict
from config.settings import AppConfig

class EmailSender:
    def __init__(self, config: AppConfig):
        self.config = config

    def send(self, content: Dict[str, str]):
        if not all([self.config.SMTP_USER,
                    self.config.SMTP_PASSWORD,
                    self.config.EMAIL_TO]):
            raise ValueError("邮件配置不完整")

        message = MIMEMultipart("alternative")
        message["Subject"] = content["subject"]
        message["From"] = self.config.EMAIL_FROM
        message["To"] = self.config.EMAIL_TO

        text_part = MIMEText(content["body"], "plain", "utf-8")
        message.attach(text_part)

        html_body = self._convert_to_html(content["body"])
        html_part = MIMEText(html_body, "html", "utf-8")
        message.attach(html_part)

        with smtplib.SMTP_SSL(self.config.SMTP_HOST,
                             self.config.SMTP_PORT) as server:
            server.login(self.config.SMTP_USER, self.config.SMTP_PASSWORD)
            server.send_message(message)

    def _convert_to_html(self, text: str) -> str:
        import re
        html = text
        html = html.replace("\n\n", "</p><p>")
        html = f"<p>{html}</p>"
        html = re.sub(r'```([\s\S]*?)```',
                     r'<pre><code>\1</code></pre>', html)
        html = re.sub(r'`([^`]+)`', r'<code>\1</code>', html)
        html = re.sub(r'━+', '<hr>', html)
        return html

    def test_connection(self) -> bool:
        try:
            with smtplib.SMTP_SSL(self.config.SMTP_HOST,
                                 self.config.SMTP_PORT) as server:
                server.login(self.config.SMTP_USER, self.config.SMTP_PASSWORD)
            return True
        except Exception:
            return False
```

from sqlalchemy import Column, Integer, String, Text, Boolean, DateTime, Date, ForeignKey, JSON
from sqlalchemy.orm import relationship, declarative_base
from datetime import datetime

Base = declarative_base()

class Note(Base):
    __tablename__ = "notes"

    id = Column(Integer, primary_key=True)
    title = Column(String(200), nullable=False)
    content = Column(Text, nullable=False)
    summary = Column(Text, default="")
    algorithm_type = Column(String(50), default="其他")
    difficulty = Column(String(20), default="中等")
    tags = Column(JSON, default=list)
    mastery_level = Column(Integer, default=0)
    review_count = Column(Integer, default=0)
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)
    last_reviewed = Column(DateTime, nullable=True)
    next_review_date = Column(Date, nullable=True)

    questions = relationship("Question", back_populates="note")
    review_records = relationship("ReviewRecord", back_populates="note")

class Question(Base):
    __tablename__ = "questions"

    id = Column(Integer, primary_key=True)
    note_id = Column(Integer, ForeignKey("notes.id"), nullable=False)
    question_type = Column(String(20), nullable=False)
    content = Column(Text, nullable=False)
    answer = Column(Text, nullable=False)
    explanation = Column(Text, default="")
    created_at = Column(DateTime, default=datetime.utcnow)

    note = relationship("Note", back_populates="questions")
    answer_records = relationship("AnswerRecord", back_populates="question")

    def get_notes_due_for_review(self, due_date: date) -> List[Note]:
        return self.session.query(Note).filter(
            Note.next_review_date <= due_date
        ).all()

        note.updated_at = datetime.now()
        self.session.commit()
        self.session.refresh(note)
        return note

    def delete(self, note_id: int):
        note = self.get_by_id(note_id)
        if note:
            self.session.delete(note)
            self.session.commit()

    def update_mastery(self, note_id: int, mastery_level: int,

from datetime import datetime, timedelta
from sqlalchemy.orm import Session
from data.models import AnswerRecord, Question

class AnswerRecordRepository:
    def __init__(self, session: Session):
        self.session = session

    def create(self, record: AnswerRecord) -> AnswerRecord:
        self.session.add(record)
        self.session.commit()
        self.session.refresh(record)
        return record

    def get_by_id(self, record_id: int) -> Optional[AnswerRecord]:
        return self.session.query(AnswerRecord).filter(
            AnswerRecord.id == record_id
        ).first()

    def get_recent_records(self, days: int) -> List[AnswerRecord]:
        cutoff_date = datetime.now() - timedelta(days=days)
        return self.session.query(AnswerRecord).filter(
            AnswerRecord.answered_at >= cutoff_date
        ).order_by(AnswerRecord.answered_at.desc()).all()

    def get_by_question_id(self, question_id: int) -> List[AnswerRecord]:
        return self.session.query(AnswerRecord).filter(
            AnswerRecord.question_id == question_id
        ).order_by(AnswerRecord.answered_at.desc()).all()

    def get_by_note_id(self, note_id: int) -> List[AnswerRecord]:
        return self.session.query(AnswerRecord).join(Question).filter(
            Question.note_id == note_id
        ).all()

    def get_statistics_by_note(self, note_id: int) -> dict:
        records = self.get_by_note_id(note_id)
        if not records:
            return {"total": 0, "correct": 0, "accuracy": 0.0}
        total = len(records)
        correct = sum(1 for r in records if r.is_correct)
        return {
            "total": total,
            "correct": correct,
            "accuracy": correct / total if total > 0 else 0.0
        }

    def delete(self, record_id: int):
        record = self.get_by_id(record_id)
        if record:
            self.session.delete(record)
            self.session.commit()

new_mastery = old_mastery + delta

import re
import json

def parse_llm_response(response: str) -> dict:
    json_pattern = r'\{[\s\S]*\}|\[[\s\S]*\]'
    match = re.search(json_pattern, response)
    if match:
        try:
            return json.loads(match.group())
        except json.JSONDecodeError:
            return {"raw": response}
    return {"raw": response}
